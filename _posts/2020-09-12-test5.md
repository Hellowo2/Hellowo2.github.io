```c
#include<stdio.h>
#include<stdlib.h>
typedef int ElemType;
typedef struct node
{
	ElemType date;
	struct node *next;
}slink;
int list(slink *head);  //函数声明

//初始化单链表
slink *initlist()
{
    slink *head;
    head=(slink *)malloc(sizeof(slink));   
    if(!head)   //分配空间失败
    {
        return 0;
    }
    head->next=NULL;
    printf("初始化成功\n");
    system("pause");
    return head;
}

//尾插法插入一个新元素
int creslink_tail(slink *head)
{
	slink *p=head;
    slink *s;
    ElemType x;
    printf("插入前单链表数据为\n");
    printf("\n");
    list(head);     //输出函数
    printf("请输入一个新元素,并插到链表的尾部\n");
    scanf("%d",&x);
    s=(slink *)malloc(sizeof(slink));
    s->date=x;
    while(p->next!=NULL)    //p指向最后一个结点
    {
        p=p->next;
    }
    s->next=p->next;  //在最后一个结点插入  
    p->next=s;
    printf("\n插入后的单链表数据为\n");
    list(head);
	printf("尾插法插入成功\n");
	system("pause");
	return 0;
}

//头插法插入一个新元素
int creslink_head(slink *head)
{
	slink *s;
    ElemType x;
    s=(slink *)malloc(sizeof(slink));
    printf("插入前单链表数据为\n");
    printf("\n");
    list(head);
    printf("请输入一个新元素,并插到链表的头部\n");
    scanf("%d",&x);
    s->date=x;
    s->next=head->next;     //在第一个结点后面插入
    head->next=s;
    printf("\n插入后的单链表数据为\n");
    list(head);
	printf("头插法插入成功\n");
	system("pause");
	return 0;
}

//删除函数
int del(slink *head,int i)      //i为locate_del传过来的参数,i的值为位序,而不是下标
{
	slink *p,*q;
    int j;
	ElemType e;
	if(i<1)     //i值不在范围内
	{
		printf("error");
		return 0;
	}
	p=head;
	for(j=0;j<i-1&&p->next!=NULL;j++)   //p指向位序为i的前一位结点
	{
		p=p->next;  
	}
	if(p->next==NULL)
	{
		printf("error");
		return 0;
	}
	q=p->next;
	p->next=q->next;
	printf("\n插入后的单链表数据为\n");
    list(head);
	free(q);
	system("pause");
	return 1;
}

//定位并删除数据
int locate_del(slink *head)
{
	slink *p;
	ElemType x;
    printf("删除前单链表数据为\n");
    printf("\n");
    list(head);
	printf("输入想要删除的数据  ");
	scanf("%d",&x);
	int i=1;
	p=head->next;
	while(p!=NULL&&p->date!=x)
	{
		i++;
		p=p->next;
	}
	if(p)
	{
        del(head,i);    //向del函数要删除结点的传位序
	}
	else
	{
		printf("找不到此数据\n");
		system("pause");
		return 0;
	}	
}


int insert(slink *head)
{
	slink *p,*s;
	int j,i;
	ElemType e;
    printf("插入前单链表数据为\n");
    printf("\n");
    list(head);
	printf("输入在第几个结点后插入以及插入的数据");
	scanf("%d%d",&i,&e);
	if(i<1)
	{
		printf("error");
		return 0;
	}
	p=head;
	for(j=0;j<i&&p!=NULL;j++)   //p指向位序为i的结点
	{
		p=p->next;
	}
	if(!p)  //i值超出范围
	{
		printf("error");
        system("pause");
		return 0;
	}
	s=(slink *)malloc(sizeof(slink));
	s->date=e;
	s->next=p->next;
	p->next=s;
    printf("\n插入后的单链表数据为\n");
    list(head);
	printf("插入成功");
	system("pause");
	return 1;
}

int list(slink *head)
{
	slink *p;
	if(head->next==NULL||head==NULL)
	{
		printf("空表\n");
		return 0;
	}
	p=head->next;
	while(p!=NULL)
	{
		printf("%d\t",p->date);
		p=p->next;
	}
    printf("\n");
	return 0;
}

int hjx()
{
	int i;
    printf("输入1初始化单链表\n");
	printf("输入2用尾插法插入一个新元素\n");
	printf("输入3用头插法插入一个新元素\n");
	printf("输入4可删除值为x的元素\n");
	printf("输入5可在第i个结点后插入元素\n");
	printf("输入6输出数据\n");
	printf("输入7释放链表\n\n");
	printf("输入想要选择的功能  ");
	scanf("%d",&i);
	return i;
}

slink *release(slink *L1)
{
	if(L1->next==NULL||L1==NULL)
	{
		printf("空表  错误");
		system("pause");
		return 0;
	}
    slink *p,*q;
    p=L1->next;
    while(p!=NULL)
    {
        q=p->next;
        free(p);
        p=q;
    }
	L1->next=NULL;
	printf("释放成功\n");
	system("pause");
	return L1;
}

int main()
{
	slink *L;
	int j;
	for(;;)
	{
		system("cls");
		switch(hjx())
		{
            case 1:
            L=initlist();
            continue;
			case 2:
			creslink_tail(L);
			continue;
			case 3:
			creslink_head(L);
			continue;
            case 4:
            locate_del(L);
			continue;
			case 5:
			insert(L);continue;
			case 6:
			list(L);
            system("pause");continue;
			case 7:
			L=release(L);continue;
		}
	}
}
```
