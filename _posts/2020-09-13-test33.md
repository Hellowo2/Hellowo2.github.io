```c
#include<stdio.h>
#include<stdlib.h>
int main()
{
    int a[5]={1,2,4,7,8};
    for(int i=0;i<5;i++)
    {
        printf("%d",*(a+i));    //*(a++)错误，只有作为形参数组的数组名才能自增和自减
    }
    system("pause");
}
```
