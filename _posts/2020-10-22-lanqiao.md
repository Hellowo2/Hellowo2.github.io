```c
#include<stdio.h>
#include<stdlib.h>
int main()
{
	int t;
	int count=0;	//统计字符2的个数
	for(int i=1;i<=2020;i++)	//遍历1-2020
	{
        int x=i;	//x为1-2020中的一个数
		while(x)	//循环判断数字x的位数是否含有2
		{
			t=x%10;
			if(t==2)
			{
				count++;
			}
			x=x/10;
		}
	}
	printf("%d",count);
    system("pause");
}
```
