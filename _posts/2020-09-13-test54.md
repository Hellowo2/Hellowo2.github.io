#include<stdio.h>
#include<stdlib.h>
typedef int ElemType;
#define M 6
#define N 6
#define maxsize 100
typedef struct 
{
    int i,j;
    ElemType x;
}tupletype;
typedef struct 
{
    int rownum;
    int colnum;
    int nznum;
    tupletype data[maxsize];
}table;

int creat(table *L,int a[M][N])
{
    int i,j;
    L->rownum=M;
    L->colnum=N;
    L->nznum=0;
    for(i=0;i<M;i++)
    {
        for(j=0;j<N;j++)
        {
            if(a[i][j]!=0)
            {
                L->data[L->nznum].i=i;
                L->data[L->nznum].j=j;
                L->data[L->nznum].x=a[i][j];
                L->nznum++;
            }
        }
    }
    system("pause");
}


int getvalue(table *L)
{
    ElemType x;
    int k=0;
    int m,n;
    printf("输入你想取的行列值(二维数组的坐标):");
    scanf("%d %d",&m,&n);
    if(m>=M||n>=N)
    {
        printf("超出三元组存数据的范围\n");
        system("pause");
        return 0;
    }
    while(k<L->nznum && m>L->data[k].i)
    {
        k++;
    }
    while(k<L->nznum&&m==L->data[k].i&&n>L->data[k].j)
    {
        k++;
    }
    if(L->data[k].i==m&&L->data[k].j==n)
    {
        x=L->data[k].x;
        printf("数据为%d\n",x);
        system("pause");
    }
    else
    {
        printf("三元组没有该数据,该数据为:0\n");
        system("pause");
    }
    return 0;
}

int putvalue(table *L)
{
    if(L->nznum==maxsize)
    {
        printf("赋值有可能出现导致数组出界\n");
        system("pause");
        return 0;
    }
    ElemType x;
    int k=0,t;
    int m,n;
    printf("请输入行列值(二维数组):\n");
    scanf("%d %d",&m,&n);
    printf("请输入你要赋值的数据:\n");
    scanf("%d",&x);
    if(m>=M||n>=N)
    {
        printf("超出三元组数据范围:\n");
        system("pause");
        return 0;
    }
    while(k<L->nznum&&m>L->data[k].i)
    {
        k++;
    }
    while(k<L->nznum&&m==L->data[k].i&&n>L->data[k].j)
    {
        k++;
    }
    if(m==L->data[k].i&&n==L->data[k].j)
    {
        L->data[k].x==x;
    }
    else
    {
        for(t=L->nznum-1;t=k;t--)
        {
            L->data[t]=L->data[t+1];
        }
        L->data[k].i=m;
        L->data[k].j=n;
        L->data[k].x=x;
        L->nznum++;
    } 
}

int putlist(table *L,int a[M][N])
{
    int i,j;
    int x,k=0;
    for(i=0;i<M;i++)
    {
        for(j=0;j<N;j++)
        {
            x=0;
            if(L->data[k].i==i&&L->data[k].j==j)
            {
                x=L->data[k].x;
                k++;
                printf("%4d",x);
                continue;
            }
            printf("%4d",x);
        }
        printf("\n");
    }
    system("pause");
}
