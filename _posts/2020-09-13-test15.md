#include<stdio.h>
#include<stdlib.h>
typedef char ElemType;
typedef struct Node
{
    ElemType data;
    struct Node *left,*right;
}BitTree;
typedef struct stack 
{
	BitTree *elements[100];
	int top;
}stack;//定义一个储存树类型地址的栈，方便遍历的时候追踪到树的地址。
stack s;//定义栈
void setnull()//初始化栈
{
	s.top =0;
}
 
void push(BitTree *temp)//入栈操作
{
	s.elements[s.top++] = temp;
}
 
BitTree *pop()//取栈顶并出栈顶
{
	return s.elements[--s.top];
}

BitTree *getpop()//取栈顶并出栈顶
{
	return s.elements[(s.top)-1];
}
int empty()//判断空栈
{
	return s.top == 0;
}
//递归查找函数
BitTree *F=NULL;    //存放双亲结点指针，插入和删除使用
BitTree *find(BitTree *T,ElemType x)
{
    if(T==NULL) //二叉树为空，返回空指针
    {
        return T;
    }
    if(T->data==x)  //找到，返回结点的指针
    {
        return T;
    }
    else
    {
        F=T;
        if(T->data>x)
        find(T->left,x);    //到左子树查找
        else
        {
            find(T->right,x);   //到右子树查找
        }
        
    }
    
}

//二叉排序树的插入
BitTree *insertbst(BitTree *T,ElemType key)
{
    BitTree *C,*s;
    C=find(T,key);  //查找
    if(C==NULL) //不存在，则插入
    {
        s=(BitTree *)malloc(sizeof(BitTree));
        s->data=key;
        s->left=s->right=NULL;
        if(F==NULL) //插入结点为跟结点
        {
            T=s;
        }
        else if(key<F->data)    //接入结点为叶子结点
        F->left=s;
        else
        {
            F->right=s;
        }   
    }
    return T;
}

//建立二叉链表算法
BitTree *CreBiTree()
{
    BitTree *bt;
    ElemType x;
    //fflush(stdin);
    scanf("%c",&x);	//#代表虚结点
    if(x=='#')
    {
        bt=NULL;
    }
    else
    {
        bt=(BitTree *)malloc(sizeof(BitTree));
        bt->data=x;
        bt->left=CreBiTree();
        bt->right=CreBiTree();
    }
    return bt;   
}

//删除函数
int deletebst(BitTree *T,ElemType key)
{
    BitTree *p,*s,*q;
    p=find(T,key);
    if(!p)
    {
        printf("不存在该结点\n");
        return 0;
    }
    if(p->left==NULL)   //左子树空，则重接其右子树
    {
        q=p;
        p=p->right;
    }
    else if(p->right==NULL) //右子树空，则重接其左子树
    {
        q=p;
        s=p->left;
    }
    else    //左右子树都不空
    {
        q=p;
        s=p->left;
        while(s->right!=NULL)   //找到p的左子树的最右下结点
        {
            q=s;
            s=s->right;
        }
        if(q!=p)    //重接q的右子树
        q->right=s->left;
        else    //重接q的左子树
        {
            q->left=s->left;
        }
        q=s;    //用s结点的值替换p结点的值
        p->data=s->data;    
    }
    if(F==NULL) //若被删结点为根节点，则将p改为根节点
    T=p;
    else if(q!=s)   //左子树为空或者右子树为空的结点p与其双亲结点重新链接
    {
        if(key<F->data)
        F->left=p;
        else
        {
            F->right=p;
        }
    }
    free(q);
    return 1;
}

//先序非递归遍历
int PreOrder(BitTree *T)
{
    BitTree *p;
    setnull();  //初始化栈
    push(T);    //根节点指针入栈
    while(!empty()) //栈为空时算法结束
    {
        p=pop();    //弹栈，p指向（子树）根结点
        while(p)
        {
            printf("%c\t",p->data); //访问当前结点
            if(p->right)    //非空的右指针入栈
            push(p->right);
            p=p->left;  //沿左指针访问，直到左指针为空为止
        }
    }
}
//中序非递归遍历
int InOrder(BitTree *T)
{
    BitTree *p;
    setnull();  //初始化栈
    p=T;    //p指向根节点
    while(p || !empty())    //若p为空且栈为空时算法结束
    {
        while (p)
        {
            push(p);
            p=p->left;  //沿左指针走，沿途经过的（子树）根结点指针入栈
        }
        p=pop();
        printf("%c\t",p->data); //访问当前结点
        p=p->right; //向右跳一步到右子树上继续进行遍历过程
    }
}

//后序遍历非递归
int PostOrder(BitTree *T)
{
    BitTree *p,*q;
    setnull();  //初始化栈
    p=T;
    q=NULL;
    while(p || !empty())
    {
        if(p!=q)
        {
            while(p)
            {
                push(p);    //p非空时，压栈
                if(p->left) //沿左指针下移
                p=p->left;
                else    //若左指针空，则沿右指针下移
                {
                    p=p->right;
                }   
            }
        }
        if(empty())
        break;
        q=getpop();
        if(q->right==p) //若q的右指针为空（p为空时）或者指向刚访问的结点
        {
            p=pop();    //弹栈并访问该结点
            printf("%c\t",p->data);   //访问当前结点
        }
        else    //否则，沿q的右指针继续遍历访问
        {
            p=q->right;
        }
        
    }
}
int IsSearchTree(const BitTree *T)  //递归遍历二叉树是否为二叉排序树
{
	if(!T)        //空二叉树情况
		return 1;
	else if(!(T->left) && !(T->right))   //左右子树都无情况
		return 1;
	else if((T->left) && !(T->right))    //只有左子树情况
	{
		if(T->left->data>T->data)
			return 0;
		else
			return IsSearchTree(T->left);
	}
	else if((T->right) && !(T->left))   //只有右子树情况
	{
		if(T->right->data<T->data)
			return 0;
		else
			return IsSearchTree(T->right);
	}
	else         //左右子树全有情况
	{                                
		if((T->left->data>T->data) || (T->right->data<T->data))
			return 0;
		else
			return ( IsSearchTree(T->left) && IsSearchTree(T->right) );
	}
}

int main()
{
    ElemType x;
    BitTree *T;
    printf("请输入要建立二叉树的结点数据(#代表虚结点)\n");
    T=CreBiTree();
    printf("先序非递归遍历二叉树的结点数据为\n");
    PreOrder(T);
    printf("\n");
    printf("中序非递归遍历二叉树的结点数据为\n");
    InOrder(T);
    printf("\n");
    printf("后序非递归遍历二叉树的结点数据为\n");
    PostOrder(T);
    printf("\n");
    if(IsSearchTree(T))
    {
        printf("该二叉树为二叉排序树\n");
        printf("对该二叉排序树进行插入操作和删除操作\n");
        printf("插入的数据为(插入的数据不能和原本的结点数据一样)\n");
        fflush(stdin);
        scanf("%c",&x);
        insertbst(T,x);
        printf("删除的数据为\n");
        fflush(stdin);
        scanf("%c",&x);
        deletebst(T,x);
        printf("该二叉树最终结果为：\n");
        printf("先序非递归遍历二叉树的结点数据为\n");
        PreOrder(T);
        printf("\n");
        printf("中序非递归遍历二叉树的结点数据为\n");
        InOrder(T);
        printf("\n");
        printf("后序非递归遍历二叉树的结点数据为\n");
        PostOrder(T);
        printf("\n");
    }
    else
    {
        printf("该二叉树不是二叉排序树\n");
    }
    system("pause");
}
