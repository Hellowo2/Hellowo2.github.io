```c
#include <stdio.h>
#include <stdlib.h>
int main()
{
    int n,i;
    printf("输入一个大于2的数");
    scanf("%d",&n);
    if(n==2)
    printf("素数");
    
    else
    {
        for(i=2;i<n;i++)
        {
            if(n%i==0)
            {
            printf("不是素数");break;
            }
        }
        if(i==n)
        printf("是素数");
    }
    system("pause");

}
```
